	      \begin{itemize}
	      	\item[(a)] $\int e^{-\frac{x}{32}} \,dx$ \\
	      	      \textbf{Answer:} 
	      	      \begin{equation}
	      	      	\begin{split}
	      	      		u = -\frac{x}{32} \\
	      	      		-32 \int e^u \,du \\
	      	      		-32 e^{u} \\
	      	      		-32 e^{\frac{x}{32}} + C \\
	      	      	\end{split}
	      	      \end{equation}
	      	\item[(b)] $\int xe^{-\frac{x}{32}} \,dx$ \\
	      	      \textbf{Answer:} 
	      	      \begin{equation}
	      	      	\begin{split}
	      	      		u = x, v' = e^{-\frac{x}{32}} \\
	      	      		-32xe^{-\frac{x}{32}} - \int -32e^{-\frac{x}{32}} \,dx \\
	      	      		-32xe^{-\frac{x}{32}} - 1024e^{-\frac{x}{32}} + C \\
	      	      	\end{split}
	      	      \end{equation}
	      	\item[(c)] $\int x^{2}e^{-\frac{x}{32}} \,dx$ \\
